result allD y = do

    where asd = ((fst allD * fst y) `div` (snd allD * snd y))

-- Завдання № 2
algor :: [(Int, Int)] -> [(Int, Int)]
algor [] = []
algor [x] = [x]
algor (x:y:xs) =
    if length (x:y:xs) == 2
        then [(sqrti, i)]
    else algor ((sqrti, i):xs)
    where
    i = (fst x * snd y) + (snd x * fst y)
    sqrti = (fst x * fst y) + (snd x * snd y * (-1))

output complex_numbers =
    if (snd (x !! 0)) < 0
        then putStrLn (show (fst (x !! 0)) ++ show (snd (x !! 0)) ++ "i "++show(x)++" "++patsyuk++"*(cos("++vitaly++") + sin("++vitaly++")*i) ")
    else putStrLn (show (fst (x !! 0)) ++ "+" ++ show (snd (x !! 0)) ++ "i " ++show(x)++" "++patsyuk++"*(cos("++vitaly++") + sin("++vitaly++")*i) ")
    where
    x = (algor complex_numbers)
    xx = x !! 0
    vitaly = take 4 $ show(atan((fromIntegral(snd xx)) / (fromIntegral(fst xx))))
    patsyuk = take 4 $ show(sqrt((fromIntegral(fst xx)^2) + (fromIntegral(snd xx)^2)))

complex_numbers = [(1, -2), (1, 3), (1, 3)]

output_complex [x] iterator = do
    putStrLn(show(iterator+1)++"-е, комплексне число: " )

output_complex (x:xs) iterator = do
    putStrLn(show(iterator+1)++"-е, комплексне число: " )
    output [x]
    output_complex (xs) (iterator+1)
